[docs]def approx_equal(a, b, approx=True):
    """Check result with tolerance.
    
    return abs(a-b) <= abs(a)*tolerance
    """

    if approx:
        if abs(a-b) > abs(a)*pow(10,-6):
            return False
        else: 
            return True
    else:
        if a != b:
            return False
        else:
            return True







[docs]def check_result(a, b):
    """Check result correctness. """

    global global_cnt
    global name_list
    if not numpy.allclose(a, b):
        print global_cnt, name_list[global_cnt]
        cnt = 0
        if a.shape.__len__()  == 1:
            for i in range(a.shape[0]):
                if ( a[i] != b[i]):
                    cnt += 1
                    if cnt <= 10:
                        print i, a[i], b[i] 
        else:
            for i in range(a.shape[0]):
                if ( a[i] != b[i]).any():
                    cnt += 1
                    if cnt <= 10:
                        print i, a[i], b[i] 
        print cnt
    global_cnt += 1
